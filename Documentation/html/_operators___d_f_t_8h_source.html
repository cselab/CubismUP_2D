100  template <int FDdir>
101  void _solve_velocity(const Layer& psi, Layer& velocity)
102  {
103  const double invH2[2] = {
104  1./psi.getH0(),
105  1./psi.getH1()
106  };
107 
108  const Real factor[2] = {
109  invH2[0]*0.5,
110  invH2[1]*0.5
111  };
112 
113  if (FDdir==0)
114  {
115  for(int dy=0; dy<psi.sizeY; dy++)
116  for(int dx=0; dx<psi.sizeX; dx++)
117  {
118  const Real psi_right = psi.read((dx + 1) % psi.sizeX, dy);
119  const Real psi_left = psi.read((dx - 1 + psi.sizeX) % psi.sizeX, dy);
120  const Real psi_top = psi.read(dx, (dy + 1) % psi.sizeY);
121  const Real psi_bottom = psi.read(dx, (dy - 1 + psi.sizeY) % psi.sizeY);
122  //const Real psi_center = psi.read(dx, dy);
123 
124  velocity(dx, dy, 0) = factor[1]*(psi_top - psi_bottom);
125  velocity(dx, dy, 1) = -factor[0]*(psi_right - psi_left);
126  }
127  }
128  else if (FDdir==1)
129  {
130  for(int dy=0; dy<psi.sizeY; dy++)
131  for(int dx=0; dx<psi.sizeX; dx++)
132  {
133  const Real psi_right = psi.read((dx + 1) % psi.sizeX, dy);
134  const Real psi_top = psi.read(dx, (dy + 1) % psi.sizeY);
135  const Real psi_center = psi.read(dx, dy);
136 
137  velocity(dx, dy, 0) = invH2[1]*(psi_top - psi_center);
138  velocity(dx, dy, 1) = -invH2[0]*(psi_right - psi_center);
139  }
140  }
141  else
142  abort();
143 
144  }
