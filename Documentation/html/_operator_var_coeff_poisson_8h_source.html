114  }
115  */
116 
117  // 2. check convergence
118  Linf = 0;
119  L1 = 0;
120  L2 = 0;
121 #pragma omp parallel for schedule(static) reduction(max:Linf) reduction(+:L1) reduction(+:L2)

154  }
