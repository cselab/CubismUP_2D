75  template <typename Lab, typename BlockType>
76  void operator()(Lab & lab, const BlockInfo& info, BlockType& o) const
77  {
78  const double dh = info.h_gridpoint;
79  const double prefactor = -.5 * dt / dh;
80 
81  for(int iy=0; iy<FluidBlock::sizeY; ++iy)
82  for(int ix=0; ix<FluidBlock::sizeX; ++ix)
83  {
84  const FluidElement& phi = lab(ix ,iy );
85  const FluidElement& phiN = lab(ix ,iy+1);
86  const FluidElement& phiS = lab(ix ,iy-1);
87  const FluidElement& phiE = lab(ix+1,iy );
88  const FluidElement& phiW = lab(ix-1,iy );
89 
90  Real pN, pS, pW, pE;
91 
92  if (step>=2)
93  {
94  // p*
95  pN = 2*phiN.p - phiN.pOld;
96  pS = 2*phiS.p - phiS.pOld;
97  pW = 2*phiW.p - phiW.pOld;
98  pE = 2*phiE.p - phiE.pOld;
99  }
100  else
101  {
102  // pN
103  pN = phiN.p;
104  pS = phiS.p;
105  pW = phiW.p;
106  pE = phiE.p;
107  }
108 
109  // divU contains the pressure correction after the Poisson solver
110  o(ix,iy).u += prefactor/rho0 * (phiE.divU - phiW.divU);
111  o(ix,iy).v += prefactor/rho0 * (phiN.divU - phiS.divU);
112 
113  // add the split explicit term
114  o(ix,iy).u += prefactor * (pE - pW) * (1./phi.rho - 1/rho0);
115  o(ix,iy).v += prefactor * (pN - pS) * (1./phi.rho - 1/rho0);
116 
117  o(ix,iy).tmp = prefactor/rho0 * (phiN.divU - phiS.divU) + prefactor * (pN - pS) * (1./phi.rho - 1/rho0);
118  }
119  }
