59  {

62 
63  for(int iy=0; iy<FluidBlock::sizeY; ++iy)
64  for(int ix=0; ix<FluidBlock::sizeX; ++ix)
65  {
66  const FluidElement& phi = lab(ix ,iy );
67  const FluidElement& phiN = lab(ix ,iy+1);
68  const FluidElement& phiS = lab(ix ,iy-1);
69  const FluidElement& phiE = lab(ix+1,iy );
70  const FluidElement& phiW = lab(ix-1,iy );
71 
72  Real pN, pS, pW, pE;
73 
74  if (step>=2)
75  {
76  // p*
77  pN = 2*phiN.p - phiN.pOld;
78  pS = 2*phiS.p - phiS.pOld;
79  pW = 2*phiW.p - phiW.pOld;
80  pE = 2*phiE.p - phiE.pOld;
81  }
82  else
83  {
84  // pN
85  pN = phiN.p;
86  pS = phiS.p;
87  pW = phiW.p;
88  pE = phiE.p;
89  }
90 
91  // divU contains the pressure correction after the Poisson solver
92  o(ix,iy).u += prefactor/rho0 * (phiE.divU - phiW.divU);
93  o(ix,iy).v += prefactor/rho0 * (phiN.divU - phiS.divU);
94 
95  // add the split explicit term
96  o(ix,iy).u += prefactor * (pE - pW) * (1./phi.rho - 1/rho0);
97  o(ix,iy).v += prefactor * (pN - pS) * (1./phi.rho - 1/rho0);
98 
99  o(ix,iy).tmp = prefactor/rho0 * (phiN.divU - phiS.divU) + prefactor * (pN - pS) * (1./phi.rho - 1/rho0);
100  }
